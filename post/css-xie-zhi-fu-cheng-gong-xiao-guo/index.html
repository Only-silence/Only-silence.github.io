css写支付成功效果

2021-01-07 2 min read

HTML结构

<view class="success">
	<view class="success_left"></view>
	<view class="success_right"></view>
</view>

CSS样式

.success {
  height: 100rpx;
  width: 100rpx;
  border-radius: 50%;
  position: relative;
  background-color: green;
}

.success view {
  content: '';
  position: absolute;
  width: 8px;
  transform-origin: left top;  /* 控制元素的旋转基点位置（不设置此属性会导致位置漂移） */
  transition: ease 0.5;  /* 过度效果 快启动，慢停止 */
  border-radius: 10px;
  background-color: #fff;
}

.success_left {
  top: 55%;
  left: 12%;
  transform: rotate(-50deg);  /* 旋转 */
  /* 贝塞尔曲线，设置动画效果 0s表示立即执行 forwards表示执行结束保持状态 */
  animation: reverseTop 0.3s cubic-bezier(0.35, 0, 0.25, 1) 0s forwards;
}

.success_right {
  top: 80%;
  left: 43%;
  transform: rotate(-135deg);  /* 旋转 */
  /* 贝塞尔曲线，设置动画效果 0.3s表示0.3秒后执行（无缝衔接） forwards表示执行结束保持状态 */
  animation: reverseBottom 0.2s cubic-bezier(0.35, 0, 0, 1) 0.3s forwards;
}

CSS动画

@keyframes reverseTop {
  0% {
      height: 0;
  }
  100% {
      height: 40rpx;
  }
}

@keyframes reverseBottom {
  0% {
      height: 0;
  }
  100% {
      height: 65rpx;
  }
}
